//78.Dutch National Flag Algorithm ( Sort 0s,1s & 2s)
#include <stdio.h>
int main() {
    int n;
    scanf("%d",&n);

    int a[n];

    for(int i=0;i<n;i++)
        scanf("%d",&a[i]);

    int low=0,mid=0,high=n-1;

    while(mid<=high) {

        if(a[mid]==0) {
            int temp=a[low];
            a[low]=a[mid];
            a[mid]=temp;
            low++;
            mid++;
        }
        else if(a[mid]==1) {
            mid++;
        }
        else {
            int temp=a[mid];
            a[mid]=a[high];
            a[high]=temp;
            high--;
        }
    }

    for(int i=0;i<n;i++)
        printf("%d ",a[i]);

    return 0;
}
